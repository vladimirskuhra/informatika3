#include <iostream>
//#include <cstdint>
#include <string>
#include <functional>
#include <vector>



int main(){
    int xs[5] = {1, 2, 3, 4, 5};
    int velkost = sizeof(xs) / sizeof(xs[0]);
    //vypis cez indexovanie pola
    std::cout << "Cez indexovanie: ";
    for (int i = 0; i < velkost; ++i) {
        std::cout << xs[i] << " ";
    }
    std::cout << std::endl;

    std::cout << "Cez ukazovatel: ";
    int *p = xs; //xs je ukazovatelom na prvy prvok pola
    for (int i = 0; i < velkost; ++i) {
        std::cout << *(p + i) << " ";
    }
    std::cout << std::endl;

    std::cout << "Dynamicke pole/vector: ";
    std::vector<int> xt = {1, 2, 3, 4, 5};
    for (size_t i = 0; i < xt.size(); ++i) {
        std::cout << xs[i] << " ";
    }
    return 0; 
}
